### 1. `resource_tools_native.gd`中的`godot://script/current`实现
```gdscript
static func _resource_script_current(params: Dictionary, editor_interface: EditorInterface) -> Dictionary:
    """读取 godot://script/current 资源"""
    if not editor_interface:
        return {"error": "EditorInterface 不可用"}
    
    # 获取脚本编辑器
    var script_editor = editor_interface.get_script_editor()
    if not script_editor:
        return {"error": "无法获取脚本编辑器"}
    
    # 获取当前打开的脚本
    var current_script = script_editor.get_current_script()
    if not current_script:
        return {"error": "当前没有打开的脚本"}
    
    # 读取脚本内容
    var script_path: String = current_script.resource_path
    if not FileAccess.file_exists(script_path):
        return {"error": "脚本文件不存在: " + script_path}
    
    var file = FileAccess.open(script_path, FileAccess.READ)
    if not file:
        return {"error": "无法打开脚本文件: " + script_path}
    
    var script_content: String = file.get_as_text()
    file.close()
    
    # 获取脚本信息
    var script_info: Dictionary = {
        "path": script_path,
        "name": current_script.get_class(),
        "content": script_content,
        "line_count": script_content.get_line_count() if script_content else 0,
        "timestamp": Time.get_unix_time_from_system()
    }
    
    return {
        "contents": [{
            "uri": "godot://script/current",
            "mimeType": "application/json",
            "text": JSON.stringify(script_info, "\t", true)
        }]
    }
```
