# 场景结构缓存
var _scene_structure_cache: Dictionary = {}
var _cache_timestamp: Dictionary = {}

func _get_scene_structure_cached(scene_path: String) -> Dictionary:
    var cache_key = scene_path
    var current_time = Time.get_time_from_system()
    
    # 检查缓存是否有效（5 分钟有效期）
    if _scene_structure_cache.has(cache_key):
        var cache_time = _cache_timestamp.get(cache_key, 0)

            return _scene_structure_cache[cache_key]
    
    # 重新计算
    var structure = _compute_scene_structure(scene_path)
    _scene_structure_cache[cache_key] = structure
    _cache_timestamp[cache_key] = current_time
    
    return structure

# 按需加载工具
var _lazy_tools: Dictionary = {}

func _register_tool_lazy(name: String, loader: Callable) -> void:
    _lazy_tools[name] = loader

func _get_tool(name: String) -> Callable:
    if _tools.has(name):
        return _tools[name].callable
    
    if _lazy_tools.has(name):
        var loader = _lazy_tools[name]
        var tool = loader.call()
        _tools[name] = tool
        _lazy_tools.erase(name)
        return tool.callable

## 3. 优化建议

### 3.1 架构优化

#### 建议 3.1.1: 使用信号解耦通信（根据 godot-dev-guide）

**现状**: 使用直接函数调用  
**优化**:
```gdscript
# 在 NativeMCPServer 中定义信号
signal tool_execution_started(tool_name: String, params: Dictionary)
signal tool_execution_completed(tool_name: String, result: Dictionary)
signal tool_execution_failed(tool_name: String, error: String)

# 在工具执行时使用信号
func _execute_tool(tool_name: String, params: Dictionary) -> Dictionary:
    tool_execution_started.emit(tool_name, params)
    
    var tool = _tools.get(tool_name)
    if not tool:
        tool_execution_failed.emit(tool_name, "Tool not found")
        return {"error": "Tool not found: " + tool_name}
    
    try:
        var result = tool.callable.call(params)
        tool_execution_completed.emit(tool_name, result)
        return result
    except:
        tool_execution_failed.emit(tool_name, "Execution failed")
        return {"error": "Tool execution failed"}
```

### 3.3 安全增强

#### 建议 3.3.1: 路径验证增强

```gdscript
const ALLOWED_PATHS: Array[String] = ["res://", "user://"]
const BLOCKED_PATTERNS: Array[String] = ["..", "~", "$", "|", ";", "`"]

func _validate_path(path: String) -> bool:
    # 检查白名单
    var is_allowed = false
    for allowed in ALLOWED_PATHS:
        if path.begins_with(allowed):
            is_allowed = true
            break
    
    if not is_allowed:
        return false
    
    # 检查黑名单模式
    for pattern in BLOCKED_PATTERNS:
        if path.contains(pattern):
            return false
    
    # 检查路径长度
    if path.length() > 4096:
        return false
    
    return true
```

#### 建议 3.3.2: 添加速率限制

```gdscript
var _request_count: Dictionary = {}  # client_id -> count
var _request_timestamps: Dictionary = {}  # client_id -> timestamps
var _rate_limit: int = 100  # 每 60 秒最多 100 个请求

func _check_rate_limit(client_id: int) -> bool:
    var current_time = Time.get_time_from_system()
    
    if not _request_timestamps.has(client_id):
        _request_timestamps[client_id] = []
        _request_count[client_id] = 0
    
    var timestamps = _request_timestamps[client_id]
    
    # 移除 60 秒前的记录
    while not timestamps.is_empty() and current_time - timestamps[0] > 60:
        timestamps.pop_front()
        _request_count[client_id] -= 1
    
    # 检查是否超过限制
    if _request_count[client_id] >= _rate_limit:
        return false
    
    # 添加新记录
    timestamps.append(current_time)
    _request_count[client_id] += 1
    
    return true
```
